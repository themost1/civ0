Type: Single
The eight surronding pixels
Remarks
This method only returns the r channel of the pixel. It does not return the full Color struct. This method is written for internal use, and a more general method that returns a Color struct can easily be constructed by viewing the source for this method.
Examples
The following code retrieves the pixels surrounding the given pixel.
Copy
class GetPixels : MonoBehaviour
{
   public Texture2D texture;
   public Vector2 pixelLocation;

   float[] surroundingPixels;

   void Start()
   {
      surroundingPixels = CivGridUtility.GetSurrondingPixels(texture, (int)pixelLocation.x, (int)pixelLocation.y);

       for (int i = 0; i < surroundingPixels.Length; i++)
       {
           Debug.Log("Pixel " + i + ": " + surroundingPixels[i]);
       }
   }
}
