Examples
The following example adds a resource, then retrieves it by it's name. Using AddResource(Resource) is not encouraged. Add resources in the inspector.
Copy
using System;
using UnityEngine;
using CivGrid;

public class ExampleClass : MonoBehaviour
{
   TileManager tileManager;

   void Start()
   {
       tileManager = GameObject.FindObjectOfType<TileManager>();

       //this method is not encouraged, used as a specific example and not best practice. Add tiles in the
       //inspector instead.
       tileManager.AddTile(new Tile("Test", 0.0f, 0.2f));

       Tile tile = tileManager.TryGetTile("Test");

       Debug.Log(tile.name);
   }
}

//Output:
//"Test"
